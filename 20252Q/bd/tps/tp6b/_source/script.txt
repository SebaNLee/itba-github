CREATE TABLE DEPARTAMENTO
(nombre varchar(10) NOT NULL, 
 jefe varchar(10),
 primary key(nombre)
);


CREATE TABLE EMPLEADO
(nombre varchar(10) NOT NULL, 
 sueldo integer,
 trabaja_en varchar(10),
 primary key(nombre),
 foreign key(trabaja_en) references DEPARTAMENTO
);

CREATE TABLE PROVEEDOR
(nombre varchar(20) NOT NULL, 
 direccion varchar(20),
 primary key(nombre)
);


CREATE TABLE ARTICULO
(codigo int NOT NULL,
 descripcion varchar(30), 
 stock integer,
 vendido_por varchar(20) check(vendido_por in ('Hardware', 'Software')),
 primary key(codigo)
);


CREATE TABLE PROVEE
(codigo int NOT NULL,
 nombre varchar(20) NOT NULL, 
precio integer,
 primary key(codigo, nombre),
 foreign key(codigo) references articulo,
 foreign key(nombre) references proveedor
);
