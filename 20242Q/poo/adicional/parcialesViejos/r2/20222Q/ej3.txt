class Team
  def initialize(team)
    @team = team
  end

  def ==(other)
    return false unless other.is_a?(Team)
    @team == other.team
  end

  def hash
    [@team].hash
  end

  alias_method :eql?, :==

  protected attr_reader :team
end

class TeamPosition

  include Comparable # ??

  def initialize(team)
    @team = team
    @points = 0
    @gf = 0
    @gc = 0
  end

  def update_position(gf, gc)
    if gf > gc
      @points += 3
    elsif gf == gc 
      @points += 1
    end
    @gc += gc
    @gf += gf
  end

  def to_s 
    "#{@team.}\t#{@points}\t#{@gf}\t#{@gc}\n"
  end
end
